 CREATE TABLE "CARROS" 
   (	"ID" NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE  NOKEEP  NOSCALE  NOT NULL ENABLE, 
	"MARCA" VARCHAR2(50), 
	"MODELO" VARCHAR2(100), 
	"ANO" NUMBER, 
	"COR" VARCHAR2(20), 
	 CONSTRAINT "CARROS_PK" PRIMARY KEY ("ID")
  USING INDEX  ENABLE
   ) ;

   COMMENT ON COLUMN "CARROS"."ID" IS 'ID CARRO';
   COMMENT ON COLUMN "CARROS"."MARCA" IS 'MARCA DO CARRO';
   COMMENT ON COLUMN "CARROS"."MODELO" IS 'MODELO CARRO';
   COMMENT ON COLUMN "CARROS"."ANO" IS 'ANO CARRO';
   COMMENT ON COLUMN "CARROS"."COR" IS 'COR CARRO';
