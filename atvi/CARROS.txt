 CREATE TABLE CARROS (	
	ID INT NOT NULL, 
	MARCA VARCHAR(250), 
	MODELO VARCHAR(100), 
	ANO INT, 
	COR VARCHAR2(20), 
	CONSTRAINT "CARROS_PK" PRIMARY KEY ("ID")
  USING INDEX  ENABLE
   ) ;

   COMMENT ON COLUMN "CARROS"."ID" IS 'ID CARRO';
   COMMENT ON COLUMN "CARROS"."MARCA" IS 'MARCA DO CARRO';
   COMMENT ON COLUMN "CARROS"."MODELO" IS 'MODELO CARRO';
   COMMENT ON COLUMN "CARROS"."ANO" IS 'ANO CARRO';
   COMMENT ON COLUMN "CARROS"."COR" IS 'COR CARRO';
