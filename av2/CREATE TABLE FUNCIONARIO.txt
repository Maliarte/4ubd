  CREATE TABLE FUNCIONARIO (	
        ID_FUNC INT NOT NULL, 
	CPF NUMBER NOT NULL, 
	RUA VARCHAR2(45), 
	SETOR VARCHAR2(45), 
	NOME VARCHAR2(45) NOT NULL, 
	NUMERO INT, 
	ESTADO VARCHAR2(2), 
	CIDADE VARCHAR2(45), 
	TELEFONE NUMBER(20,0), 
	DATA_NASC DATE, 
	CARGO VARCHAR2(45), 
	SALARIO FLOAT(126), 
	ENTRADA_SERV TIMESTAMP (6) WITH TIME ZONE, 
	SAIDA_SERV TIMESTAMP (6) WITH TIME ZONE, 
	CONSTRAINT "FUNCIONARIO_PK" PRIMARY KEY ("ID_FUNC")
  USING INDEX  ENABLE
   ) ;
